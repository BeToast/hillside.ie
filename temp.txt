import { arrayFromPath, getPage, getRoutes } from 'app/getContent';




export default function Page({ params }: { params: { s1: string, s2: string } }){

  return (
    <div>
      <h1>{`${params.s1}/${params.s2}`}</h1>
      {getPage(params.s2)}
    </div>
  );
}




export function generateStaticParams(): { slug: string[] }[] {
  const params = getRoutes(null).map((r) => ( arrayFromPath(r) ));
  console.log(params);
  return params;
}
